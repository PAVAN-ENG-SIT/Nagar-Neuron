async def classify_with_consensus(image_bytes):
    """
    Call 3 AI models and use consensus voting
    Models: Florence-2, BLIP, CLIP
    """
    # Call models in parallel
    florence_result = await call_huggingface("microsoft/Florence-2-base", image_bytes)
    blip_result = await call_huggingface("Salesforce/blip-image-captioning-large", image_bytes)
    clip_result = await call_huggingface("openai/clip-vit-base-patch32", image_bytes)
    
    # Extract categories from each
    florence_category = extract_category(florence_result)
    blip_category = extract_category(blip_result)
    clip_category = extract_category(clip_result)
    
    # Consensus logic: 2 out of 3
    categories = [florence_category, blip_category, clip_category]
    category_counts = Counter(categories)
    final_category = category_counts.most_common(1)[0][0]
    
    # Calculate confidence: 100% if all agree, 67% if 2 agree, 33% if all disagree
    max_count = category_counts.most_common(1)[0][1]
    confidence = (max_count / 3) * 100
    
    return {
        "category": final_category,
        "confidence": confidence,
        "model_responses": {
            "florence": florence_result,
            "blip": blip_result,
            "clip": clip_result
        }
    }

import imagehash
from PIL import Image

async def check_duplicates(image_bytes, location, category):
    """
    Use perceptual hashing to find duplicate images
    Check within 100m radius and last 7 days
    """
    # Compute pHash
    image = Image.open(BytesIO(image_bytes))
    phash = str(imagehash.phash(image))
    
    # Query MongoDB for similar hashes
    seven_days_ago = datetime.now() - timedelta(days=7)
    
    existing_complaints = complaints_collection.find({
        "category": category,
        "created_at": {"$gte": seven_days_ago},
        "location.gps": {
            "$near": {
                "$geometry": {"type": "Point", "coordinates": [location['lng'], location['lat']]},
                "$maxDistance": 100  # 100 meters
            }
        }
    })
    
    # Calculate Hamming distance
    for complaint in existing_complaints:
        existing_hash = imagehash.hex_to_hash(complaint['perceptual_hash'])
        current_hash = imagehash.hex_to_hash(phash)
        hamming_distance = existing_hash - current_hash
        
        if hamming_distance < 10:  # Very similar images
            return {
                "is_duplicate": True,
                "original_complaint_id": complaint['complaint_id'],
                "similarity": 100 - (hamming_distance * 10)
            }
    
    return {"is_duplicate": False, "perceptual_hash": phash}
```

async def verify_complaint(complaint_id, user_id, verification_status, photo=None):
    """
    Allow users to verify complaints with photos
    3 verifications needed for "Verified" status
    """
    verification = {
        "user_id": user_id,
        "status": verification_status,  # "yes", "no", "cant_verify"
        "photo": photo,  # Optional verification photo
        "timestamp": datetime.now()
    }
    
    # Add verification to complaint
    complaint = complaints_collection.find_one_and_update(
        {"complaint_id": complaint_id},
        {
            "$push": {"verifications": verification},
            "$inc": {"verification_count": 1}
        },
        return_document=True
    )
    
    # Count yes/no votes
    yes_votes = sum(1 for v in complaint['verifications'] if v['status'] == 'yes')
    no_votes = sum(1 for v in complaint['verifications'] if v['status'] == 'no')
    
    # Update status based on votes
    if yes_votes >= 3:
        complaints_collection.update_one(
            {"complaint_id": complaint_id},
            {"$set": {"status": "Verified", "verification_status": "verified"}}
        )
    elif no_votes >= 2:
        complaints_collection.update_one(
            {"complaint_id": complaint_id},
            {"$set": {"status": "Resolved", "verification_status": "community_verified_fixed"}}
        )
    
    # Award points to verifier
    await award_points(user_id, "verify_complaint")
    
    return {"success": True, "verification_count": yes_votes + no_votes}

import React, { useState, useEffect } from 'react';
import { View, FlatList, RefreshControl, ScrollView, Text } from 'react-native';
import { getComplaints, getNearbyUnverified } from '../utils/api';
import ComplaintCard from '../components/Cards/ComplaintCard';
import FilterBar from '../components/UI/FilterBar';
import StatCard from '../components/Cards/StatCard';
import OfflineBanner from '../components/UI/OfflineBanner';
import { useNetInfo } from '@react-native-community/netinfo';
import { useTranslation } from 'react-i18next';

export default function HomeScreen({ navigation }) {
  const { t } = useTranslation();
  const netInfo = useNetInfo();
  const [complaints, setComplaints] = useState([]);
  const [nearbyUnverified, setNearbyUnverified] = useState([]);
  const [filters, setFilters] = useState({ category: 'all', status: 'all' });
  const [refreshing, setRefreshing] = useState(false);
  const [stats, setStats] = useState({ total: 0, open: 0, resolved: 0 });
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadData();
    // Poll for updates every 30 seconds
    const interval = setInterval(loadData, 30000);
    return () => clearInterval(interval);
  }, [filters]);

  const loadData = async () => {
    try {
      const [complaintsData, nearbyData, statsData] = await Promise.all([
        getComplaints(filters),
        getNearbyUnverified(),
        getStats()
      ]);
      
      setComplaints(complaintsData);
      setNearbyUnverified(nearbyData);
      setStats(statsData);
    } catch (error) {
      console.error('Error loading data:', error);
    } finally {
      setLoading(false);
      setRefreshing(false);
    }
  };

  const onRefresh = () => {
    setRefreshing(true);
    loadData();
  };

  return (
    <View style={{ flex: 1, backgroundColor: '#f5f5f5' }}>
      {!netInfo.isConnected && <OfflineBanner />}
      
      <FlatList
        data={complaints}
        renderItem={({ item }) => (
          <ComplaintCard
            complaint={item}
            onPress={() => navigation.navigate('Detail', { id: item.complaint_id })}
          />
        )}
        keyExtractor={(item) => item.complaint_id}
        refreshControl={
          <RefreshControl refreshing={refreshing} onRefresh={onRefresh} />
        }
        ListHeaderComponent={
          <>
            {/* Stats Summary */}
            <ScrollView horizontal showsHorizontalScrollIndicator={false} style={{ padding: 16 }}>
              <StatCard title={t('total_complaints')} value={stats.total} icon="list" color="#3b82f6" />
              <StatCard title={t('open_issues')} value={stats.open} icon="alert-circle" color="#f59e0b" />
              <StatCard title={t('resolved')} value={stats.resolved} icon="check-circle" color="#10b981" />
            </ScrollView>

            {/* Nearby Unverified Complaints */}
            {nearbyUnverified.length > 0 && (
              <View style={{ padding: 16, backgroundColor: '#fef3c7', marginBottom: 8 }}>
                <Text style={{ fontSize: 16, fontWeight: 'bold', marginBottom: 8 }}>
                  {t('verify_nearby')} ğŸ¯
                </Text>
                <Text style={{ color: '#92400e' }}>
                  {nearbyUnverified.length} {t('complaints_need_verification')}
                </Text>
                <TouchableOpacity
                  onPress={() => navigation.navigate('NearbyUnverified')}
                  style={{ marginTop: 8, backgroundColor: '#f59e0b', padding: 10, borderRadius: 8 }}
                >
                  <Text style={{ color: 'white', fontWeight: 'bold', textAlign: 'center' }}>
                    {t('verify_now')}
                  </Text>
                </TouchableOpacity>
              </View>
            )}

            {/* Filters */}
            <FilterBar
              filters={filters}
              onFilterChange={setFilters}
            />
          </>
        }
        ListEmptyComponent={
          <EmptyState
            icon="inbox"
            title={t('no_complaints')}
            subtitle={t('no_complaints_subtitle')}
          />
        }
      />
    </View>
  );
}

COMPLETE IMPLEMENTATION WITH DUPLICATE DETECTION:
```typescript
import React, { useState, useEffect } from 'react';
import { View, Button, Image, Alert, ActivityIndicator, Text } from 'react-native';
import * as ImagePicker from 'expo-image-picker';
import * as Location from 'expo-location';
import { uploadComplaint } from '../utils/api';
import DuplicateWarningModal from '../components/Modals/DuplicateWarningModal';
import PointsEarnedModal from '../components/Modals/PointsEarnedModal';
import { useTranslation } from 'react-i18next';

export default function UploadScreen({ navigation }) {
  const { t } = useTranslation();
  const [image, setImage] = useState(null);
  const [location, setLocation] = useState(null);
  const [loading, setLoading] = useState(false);
  const [duplicateInfo, setDuplicateInfo] = useState(null);
  const [pointsEarned, setPointsEarned] = useState(null);

  useEffect(() => {
    requestPermissions();
  }, []);

  const requestPermissions = async () => {
    const { status: cameraStatus } = await ImagePicker.requestCameraPermissionsAsync();
    const { status: locationStatus } = await Location.requestForegroundPermissionsAsync();
    
    if (cameraStatus !== 'granted' || locationStatus !== 'granted') {
      Alert.alert(t('permissions_required'), t('permissions_message'));
    }
  };

  const takePhoto = async () => {
    const result = await ImagePicker.launchCameraAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsEditing: true,
      aspect: [4, 3],
      quality: 0.8,
      exif: true  // Include EXIF data for GPS
    });

    if (!result.canceled) {
      setImage(result.assets[0]);
      await extractLocation();
    }
  };

  const pickFromGallery = async () => {
    const result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsEditing: true,
      aspect: [4, 3],
      quality: 0.8,
      exif: true
    });

    if (!result.canceled) {
      setImage(result.assets[0]);
      await extractLocation();
    }
  };

  const extractLocation = async () => {
    try {
      const loc = await Location.getCurrentPositionAsync({
        accuracy: Location.Accuracy.High
      });
      setLocation({
        lat: loc.coords.latitude,
        lng: loc.coords.longitude
      });
    } catch (error) {
      // Fallback to default Bangalore location
      setLocation({ lat: 12.9716, lng: 77.5946 });
    }
  };

  const handleSubmit = async (forceSave = false) => {
    if (!image) {
      Alert.alert(t('no_image'), t('no_image_message'));
      return;
    }

    setLoading(true);

    try {
      const result = await uploadComplaint(image.uri, location, forceSave);

      if (result.is_duplicate && !forceSave) {
        // Show duplicate warning
        setDuplicateInfo(result.duplicate_info);
        setLoading(false);
        return;
      }

      // Show success with points earned
      setPointsEarned(result.points_earned);
      
      // Navigate back after 2 seconds
      setTimeout(() => {
        navigation.PContinuenavigate('Home');
}, 2000);
} catch (error) {
  Alert.alert(t('upload_failed'), error.message);
} finally {
  setLoading(false);
}
};
return (
<View style={{ flex: 1, padding: 16, backgroundColor: 'white' }}>
<Text style={{ fontSize: 24, fontWeight: 'bold', marginBottom: 16 }}>
{t('report_issue')}
</Text>
  {image ? (
    <Image
      source={{ uri: image.uri }}
      style={{ width: '100%', height: 300, borderRadius: 12, marginBottom: 16 }}
      resizeMode="cover"
    />
  ) : (
    <View style={{ height: 300, backgroundColor: '#f5f5f5', borderRadius: 12, justifyContent: 'center', alignItems: 'center', marginBottom: 16 }}>
      <Text style={{ color: '#9ca3af' }}>{t('no_image_selected')}</Text>
    </View>
  )}

  {location && (
    <View style={{ padding: 12, backgroundColor: '#e0f2fe', borderRadius: 8, marginBottom: 16 }}>
      <Text style={{ fontSize: 12, color: '#0369a1' }}>
        ğŸ“ {t('location')}: {location.lat.toFixed(4)}, {location.lng.toFixed(4)}
      </Text>
    </View>
  )}

  <View style={{ flexDirection: 'row', gap: 12, marginBottom: 16 }}>
    <Button title={t('take_photo')} onPress={takePhoto} />
    <Button title={t('choose_gallery')} onPress={pickFromGallery} />
  </View>

  <Button
    title={loading ? t('processing') : t('submit_complaint')}
    onPress={() => handleSubmit(false)}
    disabled={!image || loading}
    color="#3b82f6"
  />

  {loading && (
    <View style={{ position: 'absolute', top: 0, left: 0, right: 0, bottom: 0, backgroundColor: 'rgba(0,0,0,0.7)', justifyContent: 'center', alignItems: 'center' }}>
      <ActivityIndicator size="large" color="#3b82f6" />
      <Text style={{ color: 'white', marginTop: 16, fontSize: 16 }}>
        {t('analyzing_with_ai')}...
      </Text>
    </View>
  )}

  {duplicateInfo && (
    <DuplicateWarningModal
      visible={true}
      duplicateInfo={duplicateInfo}
      onContinue={() => {
        setDuplicateInfo(null);
        handleSubmit(true);
      }}
      onCancel={() => setDuplicateInfo(null)}
    />
  )}

  {pointsEarned && (
    <PointsEarnedModal
      visible={true}
      points={pointsEarned}
      onClose={() => setPointsEarned(null)}
    />
  )}
</View>
);
}
